00049     if (e != (QEvent *)0) {                                // event available?
00050 
00051         postLIFO(e);      // post it to the front of the Active Object's queue
00052 
00053         QS_CRIT_STAT_
00054         QF_CRIT_ENTRY_();
00055 
00056         if (EVT_POOL_ID(e) != (uint8_t)0) {          // is it a dynamic event?
00057 
00058             // after posting to the AO's queue the event must be referenced
00059             // at least twice: once in the deferred event queue (eq->get()
00060             // did NOT decrement the reference counter) and once in the
